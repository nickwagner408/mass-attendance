function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // Parse incoming data (supports JSON body or form-encoded) var raw = e.postData ? e.postData.contents : ''; var d; if (raw.indexOf('payload=') === 0) { d = JSON.parse(decodeURIComponent(raw.slice(8).replace(/\+/g,' '))); } else { d = JSON.parse(raw); } // Create header row if sheet is empty if (sheet.getLastRow() === 0) { sheet.appendRow(['Date','Day','Time','Mass','Adults','Children','Zoom','YouTube','Total','Recorded At']); sheet.getRange(1,1,1,10).setFontWeight('bold').setBackground('#0f1923').setFontColor('#c9a84c'); sheet.setFrozenRows(1); } // Insert new data at row 2 so newest is always on top sheet.insertRowAfter(1); sheet.getRange(2,1,1,10).setValues([[ d.date, d.day, d.time, d.massType, Number(d.adults), Number(d.children), Number(d.zoom||0), Number(d.youtube||0), Number(d.total), new Date().toLocaleString() ]]); return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error',msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:'ready'})) .setMimeType(ContentService.MimeType.JSON); }
